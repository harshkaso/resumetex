% ADD SKILL TO THE LIST
% ARGS:
% 1 SKILL CATEGORY
% 2 SKILLS {comma separated list of skills: skill1, skill2, skill3 ... }
\newcommand{\SkillList}{}
\NewDocumentCommand{\AddSkill}{ m m }{
    \g@addto@macro\SkillList{
        #1 & #2 \\
    }
}

% ADD EXPERIENCE TO THE LIST
% ARGS:
% 1 JOB POSITION
% 2 COMPANY NAME
% 3 COMPANY LOCATION
% 4 START DATE
% 5 END DATE
% 6 ACCOMPLISHMENTS {write as: \item ... \item ... }
\newcommand{\ExperienceList}{}
\NewDocumentCommand{\AddExperience}{ m m m m m +m }{
    \g@addto@macro\ExperienceList{
        \textbf{#1}, #2, #3 \hfill \textit{#4 - #5}\\
        \vspace{-7mm}
        \begin{itemize}[leftmargin=*,noitemsep, nolistsep]
            #6
        \end{itemize}
    }
}

% ADD PROJECT TO THE LIST
% ARGS:
% 1 PROJECT NAME
% 2 START DATE
% 3 END DATE
% 4 ACCOMPLISHMENTS {write as: \item ... \item ... }
% 5 COMPANY/PROJECT LINK [optional]
\newcommand{\ProjectList}{}
\NewDocumentCommand{\AddProject}{ m m m +m o }{
    \g@addto@macro\ProjectList{
        \textbf{#1}\IfNoValueTF{#5}{}{, #5} \hfill \textit{#2 - #3}\\
        \vspace{-7mm}
        \begin{itemize}[leftmargin=*,noitemsep, nolistsep]
            #4
        \end{itemize}
    }
}

% ADD EDUCATION TO THE LIST
% ARGS:
% 1 DEGREE
% 2 UNIVERSITY/SCHOOL
% 3 UNIVERISTY LOCATION
% 4 START DATE
% 5 END DATE
\newcommand{\EducationList}{}
\NewDocumentCommand{\AddEducation}{ m m m m m }{
    \g@addto@macro\EducationList{
        \textbf{#1}, #2, #3 \hfill \textit{#4 - #5}\\
    }
}
\makeatother

% DEFINE RESUME SECTION
\newenvironment{rSection}[1]{
  \textsc{\large #1} % Section title in small caps
  \medskip
  \hrule
  \begin{list}{}{ % List for each individual item in the section
    \setlength{\leftmargin}{0em} % Margin within the section
  }
  \item[]
}{
  \end{list}
}

\NewDocumentCommand{\PrintContact}{}{
    \def\DiamondSep{\quad$\diamond$\quad}
    \centerline{\huge\textbf{\MakeUppercase{\ContactName}}}

    \centerline{\ContactEmail\DiamondSep\ContactPhone\DiamondSep\ContactAddress}
}

\NewDocumentCommand{\PrintSummary}{}{
    \begin{rSection}{Summary}

}

\NewDocumentCommand{\PrintSkills}{}{

    \end{rSection}
}
